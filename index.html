// PWA Install button logic let deferredPrompt; const installAppBtn = document.getElementById('installAppBtn'); // యాప్ ఇప్పటికే ఇన్‌స్టాల్ అయి ఉంటే లేదా స్టాండ్‌అలోన్ మోడ్‌లో రన్ అవుతుంటే బటన్‌ను దాచిపెట్టండి if (window.matchMedia('(display-mode: standalone)').matches || window.navigator.standalone) { installAppBtn.classList.add('hidden'); } window.addEventListener('beforeinstallprompt', (e) => { // యాప్ ఇప్పటికే ఇన్‌స్టాల్ అయి ఉంటే, ఈవెంట్‌ను పట్టించుకోవద్దు if (window.matchMedia('(display-mode: standalone)').matches || window.navigator.standalone) { return; } e.preventDefault(); deferredPrompt = e; // బటన్‌ను ఇప్పుడు ఎనేబుల్ చేయండి మరియు స్టైల్స్ మార్చండి installAppBtn.disabled = false; installAppBtn.classList.remove('bg-gray-400'); installAppBtn.classList.add('bg-gradient-to-r', 'from-purple-600', 'to-indigo-600', 'hover:opacity-90'); }); installAppBtn.addEventListener('click', async () => { // deferredPrompt అందుబాటులో లేకపోతే ఏమీ చేయవద్దు if (!deferredPrompt) { return; } // ఇన్‌స్టాల్ ప్రాంప్ట్‌ను చూపించు deferredPrompt.prompt(); // వినియోగదారు స్పందన కోసం వేచి ఉండండి const { outcome } = await deferredPrompt.userChoice; // వినియోగదారు అంగీకరిస్తే బటన్‌ను దాచిపెట్టండి if (outcome === 'accepted') { installAppBtn.classList.add('hidden'); } deferredPrompt = null; }); window.addEventListener('appinstalled', () => { // యాప్ ఇన్‌స్టాల్ అయిన తర్వాత బటన్‌ను దాచిపెట్టండి installAppBtn.classList.add('hidden'); deferredPrompt = null; });
